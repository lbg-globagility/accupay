CREATE TABLE `calendar` (
	`RowID` INT(11) NOT NULL AUTO_INCREMENT,
	`Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`CreatedBy` INT(11) NULL DEFAULT NULL,
	`LastUpd` TIMESTAMP NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
	`LastUpdBy` INT(11) NULL DEFAULT NULL,
	`Name` VARCHAR(50) NOT NULL,
	PRIMARY KEY (`RowID`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;

CREATE TABLE `calendarday` (
	`RowID` INT(11) NOT NULL AUTO_INCREMENT,
	`Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`CreatedBy` INT(11) NULL DEFAULT NULL,
	`Updated` TIMESTAMP NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
	`UpdatedBy` INT(11) NULL DEFAULT NULL,
	`CalendarID` INT(11) NOT NULL,
	`DayTypeID` INT(11) NOT NULL,
	`Date` DATE NOT NULL,
	`Description` VARCHAR(100) NULL DEFAULT NULL,
	PRIMARY KEY (`RowID`),
	UNIQUE INDEX `CalendarID` (`CalendarID`, `Date`),
	INDEX `FK_calendarday_daytype` (`DayTypeID`),
	CONSTRAINT `FK_calendarday_calendar` FOREIGN KEY (`CalendarID`) REFERENCES `calendar` (`RowID`),
	CONSTRAINT `FK_calendarday_daytype` FOREIGN KEY (`DayTypeID`) REFERENCES `daytype` (`RowID`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;


CREATE TABLE `daytype` (
	`RowID` INT(11) NOT NULL AUTO_INCREMENT,
	`Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`CreatedBy` INT(11) NULL DEFAULT NULL,
	`Updated` TIMESTAMP NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
	`UpdatedBy` INT(11) NULL DEFAULT NULL,
	`Name` VARCHAR(100) NOT NULL,
	`DayConsideredAs` VARCHAR(50) NULL DEFAULT NULL,
	`RegularRate` DECIMAL(10,4) NOT NULL,
	`OvertimeRate` DECIMAL(10,4) NOT NULL,
	`NightDiffRate` DECIMAL(10,4) NOT NULL,
	`NightDiffOTRate` DECIMAL(10,4) NOT NULL,
	`RestDayRate` DECIMAL(10,4) NOT NULL,
	`RestDayOTRate` DECIMAL(10,4) NOT NULL,
	`RestDayNDRate` DECIMAL(10,4) NOT NULL,
	`RestDayNDOTRate` DECIMAL(10,4) NOT NULL,
	PRIMARY KEY (`RowID`),
	UNIQUE INDEX `Name` (`Name`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;
